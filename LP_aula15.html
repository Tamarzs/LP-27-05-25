1. Arquivos

Arquivos permitem armazenamento em disco.

Em C, um arquivo é um conjunto sequencial de bytes (stream). Existe uma marca de fim-de-arquivo.

2. Streams

Existem 3 stream importantes:

  • Entrada padrão (input): recebe dados do teclado.
  • Saída padrão (output): mostra dados na tela.
  • Erro padrão (error): mostra erros na tela.

2.1. Exemplo entrada padrão

Vamos criar um programa que lê um nome: 

#include <stdio.h>

int main(void) {
  char nome[20];
  scanf("%19s", nome);

  printf("Nome: %s\n", nome);
  return 0;
}

Usamos assim: 

./nome
mauro
Nome: mauro

Nesse caso a entrada vem de stdin.

Podemos trazer o nome de um arquivo entrada.txt: 

Fulano

Executando o programa assim: 

./nome < entrada.txt

Aqui stdin foi redirecionado para o arquivo entrada.txt.

2.2. Exemplo saída padrão

Agora vamos ver a saída padrão, stdout funcionando: 

#include <stdio.h>

int main(void) {
  printf("Hello, world!\n");
  return 0;
}

Normalmente, vemos a saída na tela: 

./hello

Enviando a saída para um arquivo: 

./hello > saida.txt

Aqui, stdout foi redirecionado para o arquivo saida.txt.

Outra opção é enviar a saída de erros para um arquivo: 

./hello > saida.txt 2>&1

Aqui, stderr (2) vai para stout (&1).

3. Escrevendo em arquivo

A estrutura FILE é definida em <stdio.h>. Para escrever em um arquivo abrimos o arquivo para escrita e usamos fputc() para escrever uma letra: 

#include <stdio.h>

int main(void) {
  FILE *pArquivo = NULL;

  pArquivo = fopen("nome.txt", "w");

  if( pArquivo == NULL ) {
     printf("O arquivo não foi aberto!");
  } else {
     printf("Digite um nome: ");

     char nome[20] = "";
     scanf("%19s", nome);

     for(int i = 0; i < 20; i++) {
       if( nome[i] == '\0') {
	 break;
       }

       fputc(nome[i], pArquivo);
     }
  }
  fclose(pArquivo);
  return 0;
}

4. Lendo de um arquivo

Podemos usar fgetc() para receber um char de um arquivo: 

#include <stdio.h>

int main(void) {
  char nome[20] = "";

  FILE *pArquivo = NULL;

  pArquivo = fopen("nome.txt", "r");

  if( pArquivo == NULL ) {
     printf("O arquivo não foi aberto!");
  } else {
     char letra;

     for(int i = 0; i < 20; i++) {
       letra = fgetc(pArquivo);

       nome[i] = letra;

       if( letra == '\0') {
	 break;
       }
     }
  }
  fclose(pArquivo);

  printf("Nome lido: %s\n", nome);

  return 0;
}

5. Lendo um arquivo inteiro

Podemos ler o conteúdo inteiro de um arquivo verificando se chegamos no char EOF: 

#include <stdio.h>

int main(void) {
  FILE *pArquivo = NULL;

  pArquivo = fopen("nome.txt", "r");

  if( pArquivo == NULL ) {
    printf("O arquivo não foi aberto!");
  } else {
      char nome[20] = "";
      char letra;

       for(int i = 0; i < 20; i++) {
	 letra = fgetc(pArquivo);

	 nome[i] = letra;

	 if( letra == EOF) {
	   nome[i+1] = '\0';
	   break;
	 }
       }
    }
    fclose(pArquivo);

    printf("Nome lido: %s", nome);

    return 0;
  }
